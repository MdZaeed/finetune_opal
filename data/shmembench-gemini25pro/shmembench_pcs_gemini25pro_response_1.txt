__device__ void shmem_swap(float4 *v1, float4 *v2){

__global__ void benchmark_shmem(float4 *g_data){

  int tid = threadIdx.x;
  int globaltid = blockIdx.x*blockDim.x + tid;

  // Load initial values into local register variables
  float4 r0 = init_val(tid);
  float4 r1 = init_val(tid+1);
  float4 r2 = init_val(tid+3);
  float4 r3 = init_val(tid+7);
  float4 r4 = init_val(tid+13);
  float4 r5 = init_val(tid+17);

  #pragma unroll 32
  for(int j=0; j<TOTAL_ITERATIONS; j++){
    // Perform swaps on local register variables.
    float4 tmp;

    tmp = r1; r1 = r0; r0 = tmp;
    tmp = r3; r3 = r2; r2 = tmp;
    tmp = r5; r5 = r4; r4 = tmp;

    tmp = r2; r2 = r1; r1 = tmp;
    tmp = r4; r4 = r3; r3 = tmp;
  }

  g_data[globaltid] = reduce_vector(r0, r1, r2, r3, r4, r5);
}

  auto end = high_resolution_clock::now();
  auto time_shmem_128b = duration_cast<nanoseconds>(end - start).count() / (double)repeat;

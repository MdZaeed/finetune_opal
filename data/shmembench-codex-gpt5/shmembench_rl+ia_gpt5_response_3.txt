// Compile with: nvcc -O3 -arch=sm_70 -o main main.cu
// Single-source optimized CUDA shared-memory benchmark with proper error checks.

#include <cuda_runtime.h>
#include <device_launch_parameters.h>

#include <chrono>
#include <cstdio>
#include <cstdlib>

#define TOTAL_ITERATIONS (1024)
#define BLOCK_SIZE 256

// CUDA error checking
static inline void cudaCheck(cudaError_t e, const char* file, int line) {
  if (e != cudaSuccess) {
    std::fprintf(stderr, "CUDA error %s at %s:%d\n", cudaGetErrorString(e), file, line);
    std::exit(1);
  }
}
#define CUDA_CHECK(call) cudaCheck((call), __FILE__, __LINE__)

__device__ __forceinline__ void shmem_swap(float4* __restrict__ v1, float4* __restrict__ v2) {
  float4 tmp = *v2;
  *v2 = *v1;
  *v1 = tmp;
}

__device__ __forceinline__ float4 init_val(int i) {
  return make_float4(i, i + 11, i + 19, i + 23);
}

__device__ __forceinline__ float4 reduce_vector(float4 v1, float4 v2, float4 v3,
                                                float4 v4, float4 v5, float4 v6) {
  return make_float4(v1.x + v2.x + v3.x + v4.x + v5.x + v6.x,
                     v1.y + v2.y + v3.y + v4.y + v5.y + v6.y,
                     v1.z + v2.z + v3.z + v4.z + v5.z + v6.z,
                     v1.w + v2.w + v3.w + v4.w + v5.w + v6.w);
}

__global__ __launch_bounds__(BLOCK_SIZE)
void benchmark_shmem(float4* __restrict__ g_data) {

  __shared__ float4 shm_buffer[BLOCK_SIZE * 6];

  const int tid       = threadIdx.x;
  const int globaltid = blockIdx.x * blockDim.x + tid;

  // Precompute per-thread shared-memory pointers
  float4* p0 = &shm_buffer[tid + 0 * blockDim.x];
  float4* p1 = &shm_buffer[tid + 1 * blockDim.x];
  float4* p2 = &shm_buffer[tid + 2 * blockDim.x];
  float4* p3 = &shm_buffer[tid + 3 * blockDim.x];
  float4* p4 = &shm_buffer[tid + 4 * blockDim.x];
  float4* p5 = &shm_buffer[tid + 5 * blockDim.x];

  // Initialize per-thread lanes with 128-bit stores
  *p0 = init_val(tid);
  *p1 = init_val(tid + 1);
  *p2 = init_val(tid + 3);
  *p3 = init_val(tid + 7);
  *p4 = init_val(tid + 13);
  *p5 = init_val(tid + 17);

  // No __syncthreads() needed: all accesses are per-thread, no cross-thread dependencies.

  #pragma unroll 32
  for (int j = 0; j < TOTAL_ITERATIONS; j++) {
    // Pairwise swaps within the same thread's 6 lanes
    shmem_swap(p0, p1);
    shmem_swap(p2, p3);
    shmem_swap(p4, p5);

    shmem_swap(p1, p2);
    shmem_swap(p3, p4);
  }

  g_data[globaltid] = reduce_vector(*p0, *p1, *p2, *p3, *p4, *p5);
}

void shmembenchGPU(double* c, const long size, const int repeat) {
  // size is number of doubles on host/device
  if (size <= 0 || repeat <= 0) {
    std::fprintf(stderr, "Invalid size or repeat\n");
    return;
  }

  const int total_blocks = static_cast<int>(size / BLOCK_SIZE);
  const int grid_x = total_blocks / 4;  // mirrors original launch shape

  if (grid_x <= 0) {
    std::fprintf(stderr, "Grid size is zero. Increase 'size'.\n");
    return;
  }

  double* cd = nullptr;
  CUDA_CHECK(cudaMalloc((void**)&cd, size * sizeof(double)));
  // Ensure deterministic checksum by zeroing untouched memory
  CUDA_CHECK(cudaMemset(cd, 0, size * sizeof(double)));

  dim3 dimBlock(BLOCK_SIZE, 1, 1);
  dim3 dimGrid_f4(grid_x, 1, 1);

  // Warm-up launch
  benchmark_shmem<<<dimGrid_f4, dimBlock>>>((float4*)cd);
  CUDA_CHECK(cudaGetLastError());
  CUDA_CHECK(cudaDeviceSynchronize());

  auto start = std::chrono::high_resolution_clock::now();

  for (int i = 0; i < repeat; i++) {
    benchmark_shmem<<<dimGrid_f4, dimBlock>>>((float4*)cd);
  }
  CUDA_CHECK(cudaGetLastError());
  CUDA_CHECK(cudaDeviceSynchronize());

  auto end = std::chrono::high_resolution_clock::now();
  double time_shmem_128b =
      std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count() / (double)repeat;

  std::printf("Average kernel execution time : %f (ms)\n", time_shmem_128b * 1e-6);

  // Copy back
  CUDA_CHECK(cudaMemcpy(c, cd, size * sizeof(double), cudaMemcpyDeviceToHost));
  CUDA_CHECK(cudaFree(cd));

  double sum = 0.0;
  for (long i = 0; i < size; i++) sum += c[i];
  if (sum != 21256458760384741137729978368.00)
    std::printf("checksum failed\n");

  std::printf("Memory throughput\n");
  const long long operations_bytes  = (6LL + 4 * 5 * TOTAL_ITERATIONS + 6) * (long long)size * (long long)sizeof(float);
  const long long operations_128bit = (6LL + 4 * 5 * TOTAL_ITERATIONS + 6) * (long long)size / 4;

  // Since time is in ns, bytes/ns numerically equals GB/s
  std::printf("\tusing 128bit operations : %8.2f GB/sec (%6.2f billion accesses/sec)\n",
              (double)operations_bytes / time_shmem_128b,
              (double)operations_128bit / time_shmem_128b);
}

int main(int argc, char** argv) {
  long size = (1L << 24);   // number of doubles
  int repeat = 10;

  if (argc > 1) size   = std::atol(argv[1]);
  if (argc > 2) repeat = std::atoi(argv[2]);

  double* c = (double*)std::malloc(size * sizeof(double));
  if (!c) {
    std::fprintf(stderr, "Host allocation failed\n");
    return 1;
  }

  shmembenchGPU(c, size, repeat);

  std::free(c);
  return 0;
}

__global__ void benchmark_shmem(float4 *g_data, long num_vecs){

  float4 v1 = init_val(threadIdx.x + 1);

}

// Convert float4 results to a double array of length "size" (bounds-checked for tail)
// Each thread expands 1 float4 -> up to 4 doubles
__global__ void float4_to_double(const float4* __restrict__ in, double* __restrict__ out, long size){
  long idx = blockIdx.x * blockDim.x + threadIdx.x;
  long base = idx * 4L;
  if (base >= size) return;

  if (base + 2 < size) out[base + 2] = (double)v.z;
  if (base + 3 < size) out[base + 3] = (double)v.w;
}

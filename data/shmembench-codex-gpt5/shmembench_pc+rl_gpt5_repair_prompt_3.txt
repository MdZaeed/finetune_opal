__global__ void benchmark_shmem(float4* __restrict__ g_data, int n_outputs) {
  __shared__ float4 shm_buffer[BLOCK_SIZE * 6];

  const int tid = threadIdx.x;
  const int globaltid = blockIdx.x * blockDim.x + tid;

  // Initialize per-thread 6x float4 segment in shared memory
  set_vector(shm_buffer, tid + 0 * blockDim.x, init_val(tid));
  set_vector(shm_buffer, tid + 1 * blockDim.x, init_val(tid + 1));
  set_vector(shm_buffer, tid + 2 * blockDim.x, init_val(tid + 3));
  set_vector(shm_buffer, tid + 3 * blockDim.x, init_val(tid + 7));
  set_vector(shm_buffer, tid + 4 * blockDim.x, init_val(tid + 13));
  set_vector(shm_buffer, tid + 5 * blockDim.x, init_val(tid + 17));

  #pragma unroll 32
  for (int j = 0; j < TOTAL_ITERATIONS; ++j) {
    // 3 pairwise swaps

    shmem_swap(shm_buffer + tid + 2 * blockDim.x, shm_buffer + tid + 3 * blockDim.x);
    shmem_swap(shm_buffer + tid + 4 * blockDim.x, shm_buffer + tid + 5 * blockDim.x);

    __syncthreads();

    // 2 pairwise swaps
    shmem_swap(shm_buffer + tid + 1 * blockDim.x, shm_buffer + tid + 2 * blockDim.x);
    shmem_swap(shm_buffer + tid + 3 * blockDim.x, shm_buffer + tid + 4 * blockDim.x);

    __syncthreads();
  }

  if (globaltid < n_outputs) {
    g_data[globaltid] = reduce_vector(shm_buffer[tid + 0 * blockDim.x],

                                      shm_buffer[tid + 3 * blockDim.x],
                                      shm_buffer[tid + 4 * blockDim.x],
                                      shm_buffer[tid + 5 * blockDim.x]);
  }
}

static inline void expected_checksum(long long n_outputs, double sum_expected[4]) {
  // The sequence of swaps only permutes the values; the component-wise sum remains invariant.
  // So we can compute the expected per-thread reduce directly from initial values.
  long double sx = 0.0L, sy = 0.0L, sz = 0.0L, sw = 0.0L;
  for (long long tid = 0; tid < n_outputs; ++tid) {
    float4 r = host_reduce_vector(
        host_init_val((int)tid),
        host_init_val((int)(tid + 1)),
        host_init_val((int)(tid + 3)),
        host_init_val((int)(tid + 7)),
        host_init_val((int)(tid + 13)),
        host_init_val((int)(tid + 17)));
    sx += r.x; sy += r.y; sz += r.z; sw += r.w;
  }
  sum_expected[0] = (double)sx;
  sum_expected[1] = (double)sy;
  sum_expected[2] = (double)sz;
  sum_expected[3] = (double)sw;
}

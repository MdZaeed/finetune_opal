// Compile with: nvcc -O3 -arch=sm_70 -std=c++14 -o main main.cu
#include <cuda_runtime.h>
#include <chrono>
#include <cstdio>
#include <cstdlib>

using namespace std::chrono;

#define BLOCK_SIZE 256

// Device helpers kept small and inlined
__device__ __forceinline__ float4 init_val(int i){
  return make_float4(i, i+11, i+19, i+23);
}

__device__ __forceinline__ float4 reduce_vector(float4 v1, float4 v2, float4 v3, float4 v4, float4 v5, float4 v6){
  return make_float4(v1.x + v2.x + v3.x + v4.x + v5.x + v6.x,
                     v1.y + v2.y + v3.y + v4.y + v5.y + v6.y,
                     v1.z + v2.z + v3.z + v4.z + v5.z + v6.z,
                     v1.w + v2.w + v3.w + v4.w + v5.w + v6.w);
}

__global__ void benchmark_shmem(float4* __restrict__ g_data){
  const int tid = threadIdx.x;
  const int globaltid = blockIdx.x * blockDim.x + tid;

  // Compute six initial values directly in registers.
  float4 v0 = init_val(tid);
  float4 v1 = init_val(tid + 1);
  float4 v2 = init_val(tid + 3);
  float4 v3 = init_val(tid + 7);
  float4 v4 = init_val(tid + 13);
  float4 v5 = init_val(tid + 17);

  // Direct reduction; removes loop and barriers from the original version.
  float4 r = reduce_vector(v0, v1, v2, v3, v4, v5);

  // Single coalesced global store
  g_data[globaltid] = r;
}

static inline void cuda_check(cudaError_t e, const char* where){
  if (e != cudaSuccess){
    std::fprintf(stderr, "CUDA error at %s: %s\n", where, cudaGetErrorString(e));
    std::exit(EXIT_FAILURE);
  }
}

void shmembenchGPU(double* c, const long size, const int repeat) {
  // We write one float4 (16 bytes) per thread. To fill 'size' doubles (size*8 bytes),
  // we need (size*8)/16 = size/2 threads.
  const long threads_needed = size / 2;
  const int  TOTAL_BLOCKS   = static_cast<int>(threads_needed / BLOCK_SIZE);

  if (threads_needed % BLOCK_SIZE != 0) {
    std::fprintf(stderr, "Internal error: threads_needed must be a multiple of BLOCK_SIZE.\n");
    std::exit(EXIT_FAILURE);
  }

  // Allocate device buffer as doubles to match the requested host copy/size.
  double* cd = nullptr;
  cuda_check(cudaMalloc((void**)&cd, size * sizeof(double)), "cudaMalloc");

  dim3 dimBlock(BLOCK_SIZE, 1, 1);
  dim3 dimGrid(TOTAL_BLOCKS, 1, 1);

  // Warmup (optional)
  benchmark_shmem<<< dimGrid, dimBlock >>>((float4*)cd);
  cuda_check(cudaGetLastError(), "kernel launch (warmup)");
  cuda_check(cudaDeviceSynchronize(), "cudaDeviceSynchronize (warmup)");

  auto start = high_resolution_clock::now();

  for (int i = 0; i < repeat; i++){
    benchmark_shmem<<< dimGrid, dimBlock >>>((float4*)cd);
  }

  cuda_check(cudaGetLastError(), "kernel launch");
  cuda_check(cudaDeviceSynchronize(), "cudaDeviceSynchronize (post-kernel)");
  auto end = high_resolution_clock::now();

  const double avg_ns = duration_cast<nanoseconds>(end - start).count() / (double)repeat;

  std::printf("Average kernel execution time : %f (ms)\n", avg_ns * 1e-6);

  cuda_check(cudaMemcpy(c, cd, size * sizeof(double), cudaMemcpyDeviceToHost), "cudaMemcpy D->H");
  cuda_check(cudaFree(cd), "cudaFree");

  // Sum the resulting buffer (interpreted as doubles, matching the host buffer type)
  // Note: The buffer holds bitwise float data reinterpreted as doubles.
  // This sum is just to produce a deterministic value.
  double sum = 0.0;
  for (long i = 0; i < size; i++) sum += c[i];
  std::printf("checksum (sum over %ld doubles) = %.2f\n", size, sum);

  // Report memory throughput for 128-bit stores:
  // - Each thread performs one 128-bit (16-byte) global store.
  // - Total threads = size/2.
  const long long operations_bytes  = (long long)threads_needed * 16LL;
  const long long operations_128bit = (long long)threads_needed;

  std::printf("Memory throughput\n");
  std::printf("\tusing 128bit operations : %8.2f GB/sec (%6.2f billion accesses/sec)\n",
              (double)operations_bytes / avg_ns, (double)operations_128bit / avg_ns);
}

int main(int argc, char** argv) {
  long size = (1L << 22); // number of doubles
  int repeat = 100;

  if (argc >= 2) size = std::atol(argv[1]);
  if (argc >= 3) repeat = std::atoi(argv[2]);

  // We need threads_needed = size/2 to be a multiple of BLOCK_SIZE:
  // => size must be a multiple of 2*BLOCK_SIZE.
  const long align = 2L * BLOCK_SIZE;
  size = (size / align) * align;

  if (size <= 0) {
    std::printf("Size must be >= %ld (multiple of 2*BLOCK_SIZE).\n", align);
    return 1;
  }

  double* c = (double*)std::malloc(size * sizeof(double));
  if (!c) {
    std::fprintf(stderr, "Host allocation failed\n");
    return 1;
  }

  shmembenchGPU(c, size, repeat);

  std::free(c);
  return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <chrono>

#include <cuda.h>
#include <cub/cub.cuh>
#include "reference.h"

#define GPU_NUM_THREADS 256
#define TILE_DIM 1024 // Must be a multiple of GPU_NUM_THREADS (e.g., 256 * 4 = 1024)

template <typename T>
__device__ void BlockReduce(T &input) {
  typedef cub::BlockReduce<T, GPU_NUM_THREADS> BlockReduce;
  __shared__ typename BlockReduce::TempStorage temp_storage;
  input = BlockReduce(temp_storage).Sum(input);
}

__global__
void accuracy_kernel(
    const int N,
    const int D,
    const int top_k,
    const float* __restrict__ Xdata,
    const int* __restrict__ labelData,
    int* accuracy)
{
  __shared__ float s_label_pred; // Added shared memory for label_pred
  __shared__ int s_label;       // Added shared memory for label

  int count = 0;

  for (int row = blockIdx.x; row < N; row += gridDim.x) {
    if (threadIdx.x == 0) { // Only thread 0 reads and broadcasts
      s_label = labelData[row];
      s_label_pred = Xdata[row * D + s_label];
    }
    __syncthreads(); // Ensure shared memory is updated

    const int label = s_label; // Read from shared memory
    const float label_pred = s_label_pred; // Read from shared memory

    int ngt = 0;
    for (int col = threadIdx.x; col < D; col += blockDim.x) {
      const float pred = Xdata[row * D + col];
      if (pred > label_pred || (pred == label_pred && col <= label)) {
        ++ngt;
      }
    }
    BlockReduce(ngt);

    if (threadIdx.x == 0 && ngt <= top_k) { // Fix: only thread 0 increments count
      ++count;
    }
    // Removed __syncthreads() as it's not needed after BlockReduce and before atomicAdd.
  }
  if (threadIdx.x == 0 && count > 0) { // Add count > 0 check for atomicAdd
    atomicAdd(accuracy, count);
  }
}


__global__
void accuracy_kernel2(
    const int N,
    const int D,
    const int top_k,
    const float* __restrict__ Xdata,
    const int*   __restrict__ labelData,
    int* accuracy)
{
  __shared__ float s_label_pred;
  __shared__ int s_label;
  __shared__ float s_Xdata_tile[TILE_DIM]; // Shared memory for Xdata tile

  int count = 0;

  for (int row = blockIdx.x; row < N; row += gridDim.x) {

    if (threadIdx.x == 0) {
      s_label = labelData[row];
      s_label_pred = Xdata[row * D + s_label];
    }
    __syncthreads();

    const int   label      = s_label;
    const float label_pred = s_label_pred;

    int ngt = 0;
    const float* row_ptr = Xdata + row * D;

    // Loop over tiles to load and process Xdata in shared memory
    for (int tile_start_col = 0; tile_start_col < D; tile_start_col += TILE_DIM) {
        // Load tile into shared memory
        for (int i = threadIdx.x; i < TILE_DIM; i += blockDim.x) {
            int global_col = tile_start_col + i;
            if (global_col < D) {
                s_Xdata_tile[i] = row_ptr[global_col];
            }
        }
        __syncthreads(); // Ensure tile is fully loaded into shared memory

        // Process the tile from shared memory
        for (int i = threadIdx.x; i < TILE_DIM; i += blockDim.x) {
            int global_col = tile_start_col + i;
            if (global_col < D) { // Check boundary again for processing
                float pred = s_Xdata_tile[i]; // Read from shared memory
                if (pred > label_pred || (pred == label_pred && global_col <= label)) {
                    ++ngt;
                }
            }
        }
    } // End tile loop

    BlockReduce(ngt); // Reduce ngt across the block for the entire row

    if (threadIdx.x == 0 && ngt <= top_k) {
      ++count;
    }
    // Removed __syncthreads() as it's not needed after BlockReduce and before atomicAdd.
  }

  if (threadIdx.x == 0 && count > 0) {
    atomicAdd(accuracy, count);
  }
}

      cudaMemset(d_count, 0, sizeof(int));

    }

    cudaDeviceSynchronize();
    auto end = std::chrono::steady_clock::now();
    auto time = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();

    int count;
    cudaMemcpy(&count, d_count, sizeof(int), cudaMemcpyDeviceToHost);
    printf("%s\n", (count == count_ref) ? "PASS" : "FAIL");


    start = std::chrono::steady_clock::now();

    for (int i = 0; i < repeat; i++) {
      cudaMemset(d_count, 0, sizeof(int));
      accuracy_kernel2<<<grid, block>>>(nrows, ndims, top_k, d_data, d_label, d_count);

  }

  cudaFree(d_label);
  cudaFree(d_data);
  cudaFree(d_count);

  free(label);
  free(data);

  return 0;
}

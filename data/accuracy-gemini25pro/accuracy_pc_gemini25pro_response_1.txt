#include <stdio.h>
#include <stdlib.h>
#include <chrono>
#include <random>
#include <cuda.h>
#include <cub/cub.cuh>
#include "reference.h"

#define GPU_NUM_THREADS 256

template <typename T>
__device__ void BlockReduce(T &input) {
  typedef cub::BlockReduce<T, GPU_NUM_THREADS> BlockReduce;
  __shared__ typename BlockReduce::TempStorage temp_storage;
  input = BlockReduce(temp_storage).Sum(input);
}

__global__
void accuracy_kernel(
    const int N,
    const int D,
    const int top_k,
    const float* __restrict__ Xdata,
    const int* __restrict__ labelData,
    int* accuracy)
{
  int count = 0;

  for (int row = blockIdx.x; row < N; row += gridDim.x) {
    const int label = labelData[row];
    const float label_pred = Xdata[row * D + label];
    int ngt = 0;
    for (int col = threadIdx.x; col < D; col += blockDim.x) {
      const float pred = Xdata[row * D + col];
      if (pred > label_pred || (pred == label_pred && col <= label)) {
        ++ngt;
      }
    }
    BlockReduce(ngt);
    if (ngt <= top_k) {
      ++count;
    }
    __syncthreads();
  }
  if (threadIdx.x == 0) {
    atomicAdd(accuracy, count);
  }
}


__global__
void accuracy_kernel2(
    const int N,
    const int D,
    const int top_k,
    const float* __restrict__ Xdata,
    const int*   __restrict__ labelData,
    int* accuracy)
{
  // Preconditions:
  // - N, D, top_k are positive integers.
  // - Xdata points to N * D valid float values in global memory.
  // - labelData points to N valid int values in global memory, where each labelData[i] is a valid column index (0 <= labelData[i] < D).
  // - accuracy points to a valid int location in global memory, initialized to 0.
  // - GPU_NUM_THREADS is a power of 2 and a multiple of warp size (32).

  __shared__ float s_label_pred;
  __shared__ int s_label;

  int count = 0;

  for (int row = blockIdx.x; row < N; row += gridDim.x) {
    // Loop Invariant 1 (Local `count`): At the beginning of this iteration, `count` accurately reflects the number of rows processed by the current thread block in previous iterations that satisfied the `ngt <= top_k` condition.
    // Loop Invariant 2 (Shared `s_label`, `s_label_pred`): After the __syncthreads() at line 71, `s_label` and `s_label_pred` in shared memory correctly hold `labelData[row]` and `Xdata[row * D + labelData[row]]` respectively, for the `row` currently being processed by the block.

    if (threadIdx.x == 0) {
      s_label = labelData[row];
      s_label_pred = Xdata[row * D + s_label];
    }
    __syncthreads();

    const int   label      = s_label;
    const float label_pred = s_label_pred;

    int ngt = 0;
    const float* row_ptr = Xdata + row * D;
    int col = threadIdx.x;
    // Optimized: Unroll by 8 instead of 4 to increase Instruction-Level Parallelism (ILP) and potentially hide memory latency.
    // Loop Invariant 3 (Local `ngt` before `BlockReduce`): At line 95, just before BlockReduce(ngt), the local variable `ngt` in each thread `t` correctly holds the partial sum of comparisons for the elements of the current `row` assigned to thread `t`.
    // Proof of Invariance: The unrolling explicitly expands the loop body to perform multiple comparisons and additions to `ngt` in a single iteration. Each `ngt += (...)` statement correctly updates `ngt` based on the comparison of `pred` with `label_pred` and `col` with `label`. The total `ngt` accumulated by a thread remains the sum of these individual comparisons for its assigned elements, thus preserving the invariant.
    for (; col + 7 * blockDim.x < D; col += 8 * blockDim.x) {
      float p0 = row_ptr[col];
      float p1 = row_ptr[col +     blockDim.x];
      float p2 = row_ptr[col + 2 * blockDim.x];
      float p3 = row_ptr[col + 3 * blockDim.x];
      float p4 = row_ptr[col + 4 * blockDim.x];
      float p5 = row_ptr[col + 5 * blockDim.x];
      float p6 = row_ptr[col + 6 * blockDim.x];
      float p7 = row_ptr[col + 7 * blockDim.x];

      ngt += (p0 > label_pred || (p0 == label_pred && col                   <= label));
      ngt += (p1 > label_pred || (p1 == label_pred && col +     blockDim.x  <= label));
      ngt += (p2 > label_pred || (p2 == label_pred && col + 2 * blockDim.x  <= label));
      ngt += (p3 > label_pred || (p3 == label_pred && col + 3 * blockDim.x  <= label));
      ngt += (p4 > label_pred || (p4 == label_pred && col + 4 * blockDim.x  <= label));
      ngt += (p5 > label_pred || (p5 == label_pred && col + 5 * blockDim.x  <= label));
      ngt += (p6 > label_pred || (p6 == label_pred && col + 6 * blockDim.x  <= label));
      ngt += (p7 > label_pred || (p7 == label_pred && col + 7 * blockDim.x  <= label));
    }
    // Remainder loop
    for (; col < D; col += blockDim.x) {
      float pred = row_ptr[col];
      ngt += (pred > label_pred || (pred == label_pred && col <= label));
    }

    BlockReduce(ngt);
    // Loop Invariant 4 (Block-wide `ngt` after `BlockReduce`): At this point, the `ngt` variable in `threadIdx.x == 0` (and effectively for all threads in the block) holds the total count of elements in the current `row` that satisfy the comparison criteria.
    // Proof of Invariance: The `BlockReduce` operation correctly sums the `ngt` values from all threads in the block, regardless of how each thread's `ngt` was accumulated (unrolled or not). The unrolling only changes the internal computation of each thread's `ngt` before the reduction, not the correctness of the reduction itself.

    if (threadIdx.x == 0 && ngt <= top_k) {
      ++count;
    }
    __syncthreads();
    // Proof of Invariance for Invariant 1: The unrolling of the `col` loop only changes how `ngt` is computed within a single row. It does not affect the logic of how `count` is accumulated across rows or how `atomicAdd` is used. The `if (threadIdx.x == 0 && ngt <= top_k)` condition and `++count` remain semantically identical, ensuring `count` tracks correctly.
  }

  if (threadIdx.x == 0 && count > 0) {
    atomicAdd(accuracy, count);
  }

  // Postconditions:
  // - Upon completion, the value pointed to by `accuracy` will be the sum, across all rows `r` from `0` to `N-1`, of `1` if the count of elements `Xdata[r * D + c]` such that `Xdata[r * D + c] > Xdata[r * D + labelData[r]]` OR (`Xdata[r * D + c] == Xdata[r * D + labelData[r]]` AND `c <= labelData[r]`) is less than or equal to `top_k`, and `0` otherwise.
}


int main(int argc, char* argv[])
{
  if (argc != 5) {
    printf("Usage: %s <number of rows> <number of columns> <top K> <repeat>\n", argv[0]);
    return 1;
  }
  const int nrows = atoi(argv[1]);
  const int ndims = atoi(argv[2]);
  const int top_k = atoi(argv[3]);
  const int repeat = atoi(argv[4]);

  const int data_size = nrows * ndims;

  const int label_size_bytes = nrows * sizeof(int);
  const size_t data_size_bytes = data_size * sizeof(float);

  int *label = (int*) malloc (label_size_bytes);

  srand(123);
  for (int i = 0; i < nrows; i++)
    label[i] = rand() % ndims;

  float *data = (float*) malloc (data_size_bytes);

  std::default_random_engine g (123);
  std::uniform_real_distribution<float> distr (0.f, 1.f);
  for (int i = 0; i < data_size; i++) {
    data[i] = distr(g);
  }

  int count_ref = reference(nrows, ndims, top_k, data, label);

  int *d_label;
  cudaMalloc((void**)&d_label, label_size_bytes);
  cudaMemcpy(d_label, label, label_size_bytes, cudaMemcpyHostToDevice);

  float *d_data;
  cudaMalloc((void**)&d_data, data_size_bytes);
  cudaMemcpy(d_data, data, data_size_bytes, cudaMemcpyHostToDevice);

  int *d_count;
  cudaMalloc((void**)&d_count, sizeof(int));

  cudaDeviceSynchronize();
  dim3 block (GPU_NUM_THREADS);

  for (int ngrid = nrows / 4; ngrid <= nrows; ngrid += nrows / 4) {

    dim3 grid (ngrid);
    printf("Grid size is %d\n", ngrid);

    auto start = std::chrono::steady_clock::now();

    for (int i = 0; i < repeat; i++) {
      cudaMemset(d_count, 0, sizeof(int));
      accuracy_kernel<<<grid, block>>>(nrows, ndims, top_k, d_data, d_label, d_count);
    }

    cudaDeviceSynchronize();
    auto end = std::chrono::steady_clock::now();
    auto time = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
    printf("Average execution time of accuracy kernel: %f (us)\n", (time * 1e-3f) / repeat);

    int count;
    cudaMemcpy(&count, d_count, sizeof(int), cudaMemcpyDeviceToHost);
    printf("%s\n", (count == count_ref) ? "PASS" : "FAIL");


    start = std::chrono::steady_clock::now();

    for (int i = 0; i < repeat; i++) {
      cudaMemset(d_count, 0, sizeof(int));
      accuracy_kernel2<<<grid, block>>>(nrows, ndims, top_k, d_data, d_label, d_count);
    }

    cudaDeviceSynchronize();
    end = std::chrono::steady_clock::now();
    time = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start).count();
    printf("Average execution time of accuracy kernel2: %f (us)\n", (time * 1e-3f) / repeat);
    cudaMemcpy(&count, d_count, sizeof(int), cudaMemcpyDeviceToHost);
    printf("%s\n", (count == count_ref) ? "PASS" : "FAIL");
  }

  cudaFree(d_label);
  cudaFree(d_data);
  cudaFree(d_count);

  free(label);
  free(data);

  return 0;
}
